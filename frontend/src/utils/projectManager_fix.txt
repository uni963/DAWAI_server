  // 初回起動チェック（サンプルプロジェクト読み込みが必要か）
  shouldLoadSampleProject() {
    return !this.currentProject
  }

  // サンプルプロジェクトを読み込む（互換性のため）
  loadSampleProject(sampleData) {
    if (!sampleData) return
    this.currentProject = sampleData
    this.genreContext = sampleData.genreContext || null
    this.demoSongMetadata = sampleData.demoSongMetadata || null
    this.startAutoSave()
    console.log("✅ Sample project loaded successfully")
  }
}

// シングルトンインスタンス
const projectManager = new ProjectManager()

export default projectManager
